# Plexus Agent View — backend dependencies.
#
# DEMO MODE NEEDS NOTHING HERE. server.py + demo_mode.py + events.py run on the Python
# 3.10+ stdlib alone (http.server SSE, deterministic recording replay) — no LLM key,
# no gateway, no install. That is deliberate: the demo and the e2e must run offline.
#
# The packages below are needed ONLY for LIVE mode (mode:"live"), which drives the real
# deepagent loop against a running Plexus gateway. They mirror the verified pomodoro-demo
# stack (examples/pomodoro-demo/requirements.txt), which agent_runner.py reuses.

# Live mode: the deepagent (create_deep_agent skills/tools API).
deepagents>=0.6.12,<0.7

# Live mode: the agent's "brain". langchain-anthropic is pulled in transitively by
# deepagents; langchain-openai is needed explicitly for the OpenRouter path.
langchain-openai>=1.3
langchain-anthropic>=1.4

# Live mode: HTTP client for the Plexus wire protocol (PlexusClient uses httpx.Client).
httpx>=0.27
